// TechNova Repair Booking API
const bookRepair = async (deviceInfo) => {
  const response = await fetch('/api/repairs', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      device: deviceInfo.model,
      issue: deviceInfo.problem,
      customer: deviceInfo.name,
      phone: deviceInfo.contact,
      date: deviceInfo.preferredDate
    })
  });
  
  const { bookingId, eta, technicianName } = 
    await response.json();
  
  return {
    bookingId,       // "TN-2024-08471"
    eta: `${eta}hrs`, // "2hrs"
    message: `Booking confirmed! 
    Your tech: ${technicianName}`
  };
};
